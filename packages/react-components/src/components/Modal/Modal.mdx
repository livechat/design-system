#### Example implementation

```jsx
import { GreetingQuickReply } from '@livechat/design-system-icons/react/material';

<Modal
  heading={
    <ModalHeader
      title="Modal Header"
      iconProps={{
        source: GreetingQuickReply,
        kind: 'primary',
        size: 'large',
      }}
    >
      Modal description
    </ModalHeader>
  }
  footer={
    <div>
      <Button size="medium" kind="secondary">
        Secondary
      </Button>
      <Button kind="primary" size="medium">
        Primary
      </Button>
    </div>
  }
  onClose={() => {}}
>
  Modal content
</Modal>;
```
